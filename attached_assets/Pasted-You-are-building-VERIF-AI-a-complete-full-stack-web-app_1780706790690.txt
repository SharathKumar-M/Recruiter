You are building VERIF-AI — a complete full-stack web application.
A job platform where students upload their profiles and recruiters can verify
them using AI agents that check if resumes, certificates, and GitHub profiles are authentic.

Build the ENTIRE application: frontend + backend + database + AI agents.
Everything must be fully functional and working.

==============================================================
SECTION 1 — PROJECT OVERVIEW
==============================================================

WHAT THIS APP DOES:
- Students register, upload resume + certificates + GitHub + LinkedIn + portfolio
- Recruiters post jobs (visible to students) and view who applied
- When a recruiter views a student's profile, they can trigger an AI agent
- The AI agent searches the web, scrapes GitHub, checks LinkedIn, cross-references
  certificates and resume claims, then generates a Trust Score (0–100) and a detailed
  report explaining what is authentic and what appears fake or unverifiable

TWO ROLES:
1. STUDENT — searches jobs, applies, manages their profile + documents
2. RECRUITER — posts jobs, views applicants, runs AI verification on any student

async def verify_token(creds: HTTPAuthorizationCredentials = Depends(bearer)) -> dict:
    try:
        decoded = auth.verify_id_token(creds.credentials)
        return decoded
    except Exception as e:
        raise HTTPException(status_code=401, detail=f"Invalid token: {str(e)}")

async def require_student(user: dict = Depends(verify_token)) -> dict:
    if user.get("role") != "student":
        raise HTTPException(status_code=403, detail="Students only")
    return user

async def require_recruiter(user: dict = Depends(verify_token)) -> dict:
    if user.get("role") != "recruiter":
        raise HTTPException(status_code=403, detail="Recruiters only")
    return user

@router.get("")
async def get_jobs(search: str = "", skills: str = "", roles: str = "", db=Depends(get_db)):
    query = {"is_active": True}
    if search:
        query["$or"] = [
            {"title": {"$regex": search, "$options": "i"}},
            {"company": {"$regex": search, "$options": "i"}},
            {"description": {"$regex": search, "$options": "i"}},
        ]
    if skills:
        skill_list = [s.strip() for s in skills.split(",")]
        query["skills"] = {"$in": skill_list}
    if roles:
        role_list = [r.strip() for r in roles.split(",")]
        query["job_type"] = {"$in": role_list}
    
    jobs = await db.jobs.find(query).sort("posted_at", -1).to_list(50)
    return {"success": True, "data": [serialize(j) for j in jobs]}

async def extract_resume_text(resume_url: str) -> str:
    """Download and extract text from a resume PDF URL."""
    try:
        async with httpx.AsyncClient() as client:
            response = await client.get(resume_url)
            pdf_bytes = response.content
        
        import io
        with pdfplumber.open(io.BytesIO(pdf_bytes)) as pdf:
            text = ""
            for page in pdf.pages:
                text += page.extract_text() or ""
        return text[:8000]  # limit to 8000 chars
    except Exception as e:
        return f"Could not extract resume text: {str(e)}"


def analyze_github_profile(github_url: str) -> dict:
    """Scrape and analyze a GitHub profile."""
    try:
        username = github_url.rstrip("/").split("/")[-1]
        user = gh.get_user(username)
        
        repos = list(user.get_repos())[:20]
        repo_data = []
        for repo in repos:
            if not repo.fork:
                repo_data.append({
                    "name": repo.name,
                    "description": repo.description,
                    "stars": repo.stargazers_count,
                    "language": repo.language,
                    "created": str(repo.created_at.date()),
                    "last_commit": str(repo.pushed_at.date()) if repo.pushed_at else None,
                    "commits_count": repo.get_commits().totalCount if repo.size > 0 else 0
                })
        
        return {
            "username": username,
            "name": user.name,
            "bio": user.bio,
            "company": user.company,
            "location": user.location,
            "followers": user.followers,
            "following": user.following,
            "public_repos": user.public_repos,
            "account_created": str(user.created_at.date()),
            "repos": repo_data[:10]
        }
    except Exception as e:
        return {"error": str(e), "username": github_url}


def run_verification_agent(
    student_name: str,

    github_data: dict,
    certificates: list,
    linkedin_url: Optional[str],
    portfolio_url: Optional[str]

    """
    Run the LangGraph verification agent to analyze all student data
    and produce a trust score + detailed report.
    """
    
    # Build context for the agent
    context = f"""
STUDENT PROFILE TO VERIFY:
Name: {student_name}

{resume_text}

=== GITHUB PROFILE ===
{json.dumps(github_data, indent=2)}

=== CERTIFICATES ===
{json.dumps(certificates, indent=2)}

=== LINKEDIN URL ===
{linkedin_url or 'Not provided'}

=== PORTFOLIO URL ===
{portfolio_url or 'Not provided'}
"""
    
    VERIFICATION_PROMPT = """You are VERIF-AI, an expert AI agent specialized in verifying student profiles for recruiters.

Your job is to thoroughly verify the authenticity of a student's claims by:
1. Analyzing their resume text for claims (education, work experience, skills, projects)
2. Cross-referencing GitHub activity with skills claimed in resume
3. Searching the web to verify companies, colleges, certifications mentioned
4. Checking if certificates are from legitimate institutions
5. Looking for inconsistencies between different sources
6. Detecting signs of fake or inflated claims

You have access to:
- web_search: Search the internet for verification

STUDENT DATA TO VERIFY:
{context}

VERIFICATION METHODOLOGY:
- For each company/employer mentioned: search web to verify it exists and the role is plausible
- For each educational institution: verify it exists and offers the mentioned degree
- For certificates: verify the issuing organization is legitimate and offers that certificate
- For skills claimed: cross-reference with GitHub repos (languages used, project complexity)
- For GitHub activity: check account age, commit frequency, repo authenticity
- Look for: dates that don't add up, claimed skills not reflected in GitHub, certificates from non-existent organizations, copy-pasted project descriptions

TOOLS AVAILABLE:
{tools}

Use the following format:
Thought: What do I need to verify?
Action: web_search
Action Input: specific search query
Observation: search results
... (repeat Thought/Action/Observation as needed)
Thought: I now have enough information to generate the final report
Final Answer: [JSON with the verification results]

IMPORTANT: Your Final Answer must be ONLY a valid JSON object in this exact format:
{{
  "trust_score": <number 0-100>,
  "verdict": "<AUTHENTIC | LIKELY_AUTHENTIC | SUSPICIOUS | FAKE>",
  "summary": "<2-3 sentence overall summary>",
  "strengths": ["<verified claim 1>", "<verified claim 2>"],
  "red_flags": ["<suspicious item 1>", "<suspicious item 2>"],
  "education_check": {{

    "detail": "<what was found>"
  }},

    "status": "<VERIFIED | SUSPICIOUS | FAKE>",
    "detail": "<certificate verification findings>"
  }},
  "recommendation": "<specific advice for recruiter>"

{agent_scratchpad}"""

    tools = [tavily_search]
    
    prompt = PromptTemplate.from_template(VERIFICATION_PROMPT)
    
    agent = create_react_agent(llm, tools, prompt)

        agent=agent,
        tools=tools,
        verbose=True,
        max_iterations=8,
        handle_parsing_errors=True
    )
    
    try:
        result = executor.invoke({"context": context})
        output = result.get("output", "{}")
        
        # Extract JSON from output
        json_match = re.search(r'\{.*\}', output, re.DOTALL)

from fastapi import APIRouter, Depends, HTTPException, BackgroundTasks
from app.core.database import get_db
from app.core.firebase import require_recruiter
from app.services.ai_agent import extract_resume_text, analyze_github_profile, run_verification_agent
from datetime import datetime
import asyncio

router = APIRouter(prefix="/ai", tags=["ai"])

@router.post("/analyze/{student_uid}")
async def analyze_student(
    student_uid: str,
    user: dict = Depends(require_recruiter),
    db=Depends(get_db)
):
    student = await db.users.find_one({"firebase_uid": student_uid, "role": "student"}, {"_id": 0})
    if not student:
        raise HTTPException(404, "Student not found")
    
    # Extract resume text
    resume_text = "No resume uploaded"
    if student.get("resume_url"):
        resume_text = await extract_resume_text(student["resume_url"])
    
    # Analyze GitHub
    github_data = {"error": "No GitHub URL provided"}
    if student.get("github_url"):
        github_data = analyze_github_profile(student["github_url"])
    
    # Run AI verification agent (sync in thread to not block event loop)
    loop = asyncio.get_event_loop()
    result = await loop.run_in_executor(
        None,
        run_verification_agent,
        student.get("display_name", "Unknown"),
        resume_text,
        github_data,
        student.get("certificates", []),
        student.get("linkedin_url"),
        student.get("portfolio_url")
    )
    
    # Save result to database
    verification_doc = {
        "student_uid": student_uid,
        "recruiter_uid": user["uid"],
        "result": result,
        "analyzed_at": datetime.utcnow()
    }
    await db.verifications.insert_one(verification_doc)
    
    return {"success": True, "data": result}

@router.get("/result/{student_uid}")
async def get_latest_result(student_uid: str, user: dict = Depends(require_recruiter), db=Depends(get_db)):
    result = await db.verifications.find_one(
        {"student_uid": student_uid},
        {"_id": 0},
        sort=[("analyzed_at", -1)]
    )
    if not result:
        return {"success": True, "data": None}
    return {"success": True, "data": result["result"]}
